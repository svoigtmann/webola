\newcommand{\target}[1]{%
  \begin{tikzpicture}[line width=3mm,line cap=round, line join=round,
    evaluate={
      \a=45;\b=35;\r=4.25;\d=1;\w=0.75;
      \x=3;\y=-2.25;\c=2;\l=0.7;\s=1.1;
    }]
    \foreach \r/\f in {3/#1,2/white,1/#1} {
      \draw [line width=3mm,fill=\f] (0,0) circle (\r);
    }
    \draw [#1,line width=  5mm,line cap=rect](0,0) -- ++(\a:1.45*\c);
    \draw [white  ,line width=5.5mm,line cap=rect](\a:.7*\c) -- (\a:0.78*\c);
    \draw [<-](0,0) -- ++(\a:\r)
    -- ++(\a+\b:\w) -- ++(180+\a:\d) -- ++(180+\a+\b:\w)
    -- ++(\a-\b:\w) -- ++(    \a:\d) -- ++(180+\a-\b:\w);
    %% 
    \foreach \a in {90,50} {
      \draw [line width=2mm,fill=white] (\x,\y) ++(\a:\c+\l/2) ++(\a+90:\s/2) -- ++(\a-90:\s) --++(\a:\l/2)--++(\a+90:\s)--++(\a:-\l/2);
      \draw [line width=2mm,fill=white] (\x,\y) ++(\a:\c-\l/2) ++(\a+90:\l/2) -- ++(\a-90:\l) --++(\a:\l  )--++(\a+90:\l)--++(\a:-\l/2);
    }
    \draw [fill=gray!25] (\x,\y) circle (\c);
    \foreach \a in {0,30,...,360} {
      \draw (\x,\y) ++(\a:0.66*\c) -- ++(\a:0.2*\c);
    }
    \draw (\x,\y) circle (1mm) edge [->] ++(15:0.66*\c);
  \end{tikzpicture}
}
